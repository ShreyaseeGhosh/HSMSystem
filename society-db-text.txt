Table towers{
  tower_id int [pk]
  tower_name String
}

Table residences{
  residences_id int [pk]
  floor_no int 
  flat_no int
  tower_ref int [ref: > towers.tower_id]
}


Table auth{
  auth_id int [pk]
  email String
  password String
  roles user_roles
}

Table users{
  user_id int [pk]
  first_name String
  last_name String
  gender genders
  phone_no String
  residence_details int [ref: > residences.residences_id]
  auth_details int [ref: > auth.auth_id]  
  parking_used int [ref: > parkings.parking_id]
}

Table parkings{
  parking_id int [pk]
  parking_no int
  vehicle_no String
  vehicle_type vehicleType
}

Table issues{
  issue_id int [pk]
  issue_name String
  raised_by int [ref: > users.user_id]
  issue_stage issueStage
}

Table admin{
  admin_id int [pk]
  first_name String
  last_name String
  gender genders
  phone_no String
  address String
  auth_details int [ref: > auth.auth_id]
}

Table manager{
  manager_id int [pk]
  first_name String
  last_name String
  gender genders
  phone_no String
  address String
  auth_details int [ref: > auth.auth_id]
}

Table watchmen{
  watchmen_id int [pk]
  first_name String
  last_name String
  gender genders
  phone_no String
  address String
  auth_details int [ref: > auth.auth_id]
  guards_at int [ref: > towers.tower_id]
}

Table visitors{
  visitor_id int [pk]
  names String
  parking_used int [ref: > parkings.parking_id]
  residence_visited int [ref: > residences.residences_id]
}

Table residence_helper{
  res_helper_id int [pk]
  first_name String
  last_name String
  gender genders
  address String
  residences_working int [ref: > residences.residences_id]
}
Table society_workers{
  soc_worker_id int [pk]
  first_name String
  last_name String
  gender genders
  address String
  tower_working int [ref: > towers.tower_id]
}

Table bills{
  bill_id int [pk]
  bill_for String
  amount long
  payee int [ref: > residences.residences_id] 
  status billStage
  paid_on datetime
}
// -----------------------

Enum user_roles{
  residents
  supervisior
  watchmen
  manager
}

Enum genders{
  MALE
  FEMALE
  OTHERS
}

Enum vehicleType{
  TWO_WHEELER
  FOUR_WHEELER
  OTHERS
}

Enum issueStage{
  GENERATED
  SOLVED
  RESOLVING
}

Enum billStage{
  GENERATED
  PAID
}
